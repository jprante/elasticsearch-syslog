092    @Inject
093    public SyslogService(Settings settings, Client client, NetworkService networkService) {
094        super(settings);
095        this.client = client;
096        this.networkService = networkService;
097        this.host = componentSettings.get("host");
098        this.port = componentSettings.get("port", "9500-9600");
099        this.bulkActions = componentSettings.getAsInt("bulk_actions", 1000);
100        this.bulkSize = componentSettings.getAsBytesSize("bulk_size", new ByteSizeValue(5, ByteSizeUnit.MB));
101        this.flushInterval = componentSettings.getAsTime("flush_interval", TimeValue.timeValueSeconds(5));
102        this.concurrentRequests = componentSettings.getAsInt("concurrent_requests", 4);
103        this.receiveBufferSize = componentSettings.getAsBytesSize("receive_buffer_size", new ByteSizeValue(10, ByteSizeUnit.MB));
104        this.receiveBufferSizePredictorFactory =
105                new FixedReceiveBufferSizePredictorFactory(componentSettings.getAsBytesSize("receive_predictor_size", receiveBufferSize).bytesAsInt());
106        this.index = componentSettings.get("index", "'syslog-'YYYY.MM.dd");
107        this.isTimeWindow = componentSettings.getAsBoolean("index_is_timewindow", true);
108        if (isTimeWindow) {
109            formatter = DateTimeFormat.forPattern(index);
110        }
111        this.type = componentSettings.get("type", "syslog");
112        this.messageParser = new MessageParser();
113        logger.info("syslog server at host [{}], port [{}], bulk_actions [{}], bulk_size [{}], flush_interval [{}], concurrent_requests [{}], index [{}], type [{}]",
114                host, port, bulkActions, bulkSize, flushInterval, concurrentRequests, index, type);
115    }

117    @Override
118    protected void doStart() throws ElasticsearchException {
119        bulkProcessor = BulkProcessor.builder(client, new BulkListener())
120                .setBulkActions(bulkActions)
121                .setBulkSize(bulkSize)
122                .setFlushInterval(flushInterval)
123                .setConcurrentRequests(concurrentRequests)
124                .build();
125        initializeUDP();
126        initializeTCP();
127    }
128
129    @Override
130    protected void doStop() throws ElasticsearchException {
131        if (udpChannel != null) {
132            udpChannel.close().awaitUninterruptibly();
133        }
134        if (udpBootstrap != null) {
135            udpBootstrap.releaseExternalResources();
136        }
137        if (tcpChannel != null) {
138            tcpChannel.close().awaitUninterruptibly();
139        }
140        if (tcpBootstrap != null) {
141            tcpBootstrap.releaseExternalResources();
142        }
143        bulkProcessor.close();
144    }
145
146    @Override
147    protected void doClose() throws ElasticsearchException {
148    }

150    private void initializeUDP() {
151        udpBootstrap = new ConnectionlessBootstrap(new NioDatagramChannelFactory(Executors.newCachedThreadPool(daemonThreadFactory(settings, "syslog_udp_worker"))));
152        udpBootstrap.setOption("receiveBufferSize", receiveBufferSize.bytesAsInt());
153        udpBootstrap.setOption("receiveBufferSizePredictorFactory", receiveBufferSizePredictorFactory);
154        udpBootstrap.setOption("broadcast", "false");
155        udpBootstrap.setPipelineFactory(new ChannelPipelineFactory() {
156            @Override
157            public ChannelPipeline getPipeline() throws Exception {
158                return Channels.pipeline(new Handler("udp"));
159            }
160        });
161
162        InetAddress address;
163        try {
164            address = networkService.resolveBindHostAddress(host);
165        } catch (IOException e) {
166            logger.warn("failed to resolve host {}", e, host);
167            return;
168        }
169        final InetAddress hostAddress = address;
170        PortsRange portsRange = new PortsRange(port);
171        final AtomicReference<Exception> lastException = new AtomicReference<>();
172        boolean success = portsRange.iterate(new PortsRange.PortCallback() {
173            @Override
174            public boolean onPortNumber(int portNumber) {
175                try {
176                    udpChannel = udpBootstrap.bind(new InetSocketAddress(hostAddress, portNumber));
177                } catch (Exception e) {
178                    lastException.set(e);
179                    return false;
180                }
181                return true;
182            }
183        });
184        if (!success) {
185            logger.warn("failed to bind to {}/{}", lastException.get(), hostAddress, port);
186            return;
187        }
188        logger.info("UDP listener running, address {}", udpChannel.getLocalAddress());
189    }
190
191    private void initializeTCP() {
192        tcpBootstrap = new ServerBootstrap(new NioServerSocketChannelFactory(
193                Executors.newCachedThreadPool(daemonThreadFactory(settings, "syslog_tcp_boss")),
194                Executors.newCachedThreadPool(daemonThreadFactory(settings, "syslog_tcp_worker")),
195                componentSettings.getAsInt("tcp.worker", 4)));
196        tcpBootstrap.setOption("receiveBufferSize", receiveBufferSize.bytesAsInt());
197        tcpBootstrap.setOption("receiveBufferSizePredictorFactory", receiveBufferSizePredictorFactory);
198        tcpBootstrap.setOption("reuseAddress", componentSettings.getAsBoolean("tcp.reuse_address", true));
199        tcpBootstrap.setOption("tcpNoDelay", componentSettings.getAsBoolean("tcp.no_delay", true));
200        tcpBootstrap.setOption("keepAlive", componentSettings.getAsBoolean("tcp.keep_alive", true));
201
202        tcpBootstrap.setOption("child.receiveBufferSize", receiveBufferSize.bytesAsInt());
203        tcpBootstrap.setOption("child.receiveBufferSizePredictorFactory", receiveBufferSizePredictorFactory);
204        tcpBootstrap.setOption("child.reuseAddress", componentSettings.getAsBoolean("tcp.reuse_address", true));
205        tcpBootstrap.setOption("child.tcpNoDelay", componentSettings.getAsBoolean("tcp.no_delay", true));
206        tcpBootstrap.setOption("child.keepAlive", componentSettings.getAsBoolean("tcp.keep_alive", true));
207
208        tcpBootstrap.setPipelineFactory(new ChannelPipelineFactory() {
209            @Override
210            public ChannelPipeline getPipeline() throws Exception {
211                return Channels.pipeline(new Handler("tcp"));
212            }
213        });
214
215        InetAddress address;
216        try {
217            address = networkService.resolveBindHostAddress(host);
218        } catch (IOException e) {
219            logger.warn("failed to resolve host {}", e, host);
220            return;
221        }
222        final InetAddress hostAddress = address;
223        PortsRange portsRange = new PortsRange(port);
224        final AtomicReference<Exception> lastException = new AtomicReference<>();
225        boolean success = portsRange.iterate(new PortsRange.PortCallback() {
226            @Override
227            public boolean onPortNumber(int portNumber) {
228                try {
229                    tcpChannel = tcpBootstrap.bind(new InetSocketAddress(hostAddress, portNumber));
230                } catch (Exception e) {
231                    lastException.set(e);
232                    return false;
233                }
234                return true;
235            }
236        });
237        if (!success) {
238            logger.warn("failed to bind to {}/{}", lastException.get(), hostAddress, port);
239            return;
240        }
241        logger.info("TCP listener running, address {}", tcpChannel.getLocalAddress());
242    }

244    class Handler extends SimpleChannelUpstreamHandler {
245
246        private final String protocol;
247
248        Handler(String protocol) {
249            this.protocol = protocol;
250        }
251
252        @Override
253        public void messageReceived(ChannelHandlerContext ctx, MessageEvent e) throws Exception {
254            ChannelBuffer buffer = (ChannelBuffer) e.getMessage();
255            XContentBuilder builder = jsonBuilder();
256            parse(ctx, buffer, builder);
257            IndexRequest indexRequest = new IndexRequest(isTimeWindow ? formatter.print(new DateTime()) : index)
258                    .type(type)
259                    .opType(IndexRequest.OpType.INDEX)
260                    .source(builder);
261            try {
262                bulkProcessor.add(indexRequest);
263            } catch (Exception e1) {
264                logger.warn("failed to execute bulk request", e1);
265            }
266        }
267
268        @Override
269        public void exceptionCaught(ChannelHandlerContext ctx, ExceptionEvent e) throws Exception {
270            if (e.getCause() instanceof BindException) {
271                // ignore, this happens when we retry binding to several ports, its fine if we fail...
272                return;
273            }
274            logger.warn("failure caught", e.getCause());
275        }
276
277        private void parse(ChannelHandlerContext ctx, ChannelBuffer buffer, XContentBuilder builder) throws IOException {
278            SocketAddress localAddress = ctx.getChannel().getLocalAddress();
279            SocketAddress remoteAddress = ctx.getChannel().getRemoteAddress();
280            ChannelBufferBytesReference ref = new ChannelBufferBytesReference(buffer);
281            try {
282                builder.startObject();
283                builder.field("protocol", protocol);
284                if (localAddress != null) {
285                    builder.field("local", localAddress.toString());
286                }
287                if (remoteAddress != null) {
288                    builder.field("remote", remoteAddress.toString());
289                }
290                messageParser.parseMessage(ref.toUtf8(), builder);
291                builder.endObject();
292            } catch (Exception e) {
293                logger.error(e.getMessage(), e);
294            }
295        }
296    }

298    class BulkListener implements BulkProcessor.Listener {
299
300        @Override
301        public void beforeBulk(long executionId, BulkRequest request) {
302            if (logger.isTraceEnabled()) {
303                logger.trace("[{}] executing [{}]/[{}]", executionId, request.numberOfActions(), new ByteSizeValue(request.estimatedSizeInBytes()));
304            }
305        }
306
307        @Override
308        public void afterBulk(long executionId, BulkRequest request, BulkResponse response) {
309            if (logger.isTraceEnabled()) {
310                logger.trace("[{}] executed  [{}]/[{}], took [{}]", executionId, request.numberOfActions(), new ByteSizeValue(request.estimatedSizeInBytes()), response.getTook());
311            }
312            if (response.hasFailures()) {
313                logger.warn("[{}] failed to execute bulk request: {}", executionId, response.buildFailureMessage());
314            }
315        }
316
317        @Override
318        public void afterBulk(long executionId, BulkRequest request, Throwable e) {
319            logger.warn("[{}] failed to execute bulk request", e, executionId);
320        }
321    }
