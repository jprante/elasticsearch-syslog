001package org.elasticsearch.plugin.syslog;
002
003import org.elasticsearch.common.component.LifecycleComponent;
004import org.elasticsearch.common.inject.Module;
005import org.elasticsearch.common.settings.Settings;
006import org.elasticsearch.plugins.AbstractPlugin;
007import org.elasticsearch.plugin.syslog.SyslogModule;
008import org.elasticsearch.plugin.syslog.SyslogService;
009
010import java.util.Collection;
011
012import static org.elasticsearch.common.collect.Lists.newArrayList;
013
014/**
015 * Syslog plugin
016 */
017public class SyslogPlugin extends AbstractPlugin {
018
019    private final Settings settings;
020
021    public SyslogPlugin(Settings settings) {
022        this.settings = settings;
023    }
024
025    @Override
026    public String name() {
027        return "syslog-"
028                + Build.getInstance().getVersion() + "-"
029                + Build.getInstance().getShortHash();
030    }
031
032    @Override
033    public String description() {
034        return "Syslog plugin";
035    }
036
037    @Override
038    public Collection<Class<? extends Module>> modules() {
039        Collection<Class<? extends Module>> modules = newArrayList();
040        if (settings.getAsBoolean("syslog.enabled", true)) {
041            modules.add(SyslogModule.class);
042        }
043        return modules;
044    }
045
046    @Override
047    public Collection<Class<? extends LifecycleComponent>> services() {
048        Collection<Class<? extends LifecycleComponent>> services = newArrayList();
049        if (settings.getAsBoolean("syslog.enabled", true)) {
050            services.add(SyslogService.class);
051        }
052        return services;
053    }
054
055}
