001package org.elasticsearch.common.syslog;
002
003import java.util.Comparator;
004import java.util.HashMap;
005import java.util.Map;
006
007/**
008 * Syslog severity as defined in <a href="https://tools.ietf.org/html/rfc5424">RFC 5424 - The Syslog Protocol</a>.
009 */
010public enum Severity {
011    /**
012     * Emergency: system is unusable, numerical code 0.
013     */
014    EMERGENCY(0, "EMERGENCY"),
015    /**
016     * Alert: action must be taken immediately, numerical code 1.
017     */
018    ALERT(1, "ALERT"),
019    /**
020     * Critical: critical conditions, numerical code 2.
021     */
022    CRITICAL(2, "CRITICAL"),
023    /**
024     * Error: error conditions, numerical code 3.
025     */
026    ERROR(3, "ERROR"),
027    /**
028     * Warning: warning conditions, numerical code 4.
029     */
030    WARNING(4, "WARNING"),
031    /**
032     * Notice: normal but significant condition, numerical code 5.
033     */
034    NOTICE(5, "NOTICE"),
035    /**
036     * Informational: informational messages, numerical code 6.
037     */
038    INFORMATIONAL(6, "INFORMATIONAL"),
039    /**
040     * Debug: debug-level messages, numerical code 7.
041     */
042    DEBUG(7, "DEBUG");
043
044    private final static Map<String, Severity> severityFromLabel = new HashMap<String, Severity>();
045
046    private final static Map<Integer, Severity> severityFromNumericalCode = new HashMap<Integer, Severity>();
047
048    static {
049        for (Severity severity : Severity.values()) {
050            severityFromLabel.put(severity.label, severity);
051            severityFromNumericalCode.put(severity.numericalCode, severity);
052        }
053    }
054
055    private final int numericalCode;
056
057    private final String label;
058
059    private Severity(int numericalCode, String label) {
060        this.numericalCode = numericalCode;
061        this.label = label;
062    }
063
064    /**
065     * @param numericalCode Syslog severity numerical code
066     * @return Syslog severity, not {@code null}
067     * @throws IllegalArgumentException the given numericalCode is not a valid Syslog severity numerical code
068     */
069    public static Severity fromNumericalCode(int numericalCode) throws IllegalArgumentException {
070        Severity severity = severityFromNumericalCode.get(numericalCode);
071        if (severity == null) {
072            throw new IllegalArgumentException("Invalid severity '" + numericalCode + "'");
073        }
074        return severity;
075    }
076
077    /**
078     * @param label Syslog severity textual code. {@code null} or empty returns {@code null}
079     * @return Syslog severity, {@code null} if given value is {@code null}
080     * @throws IllegalArgumentException the given value is not a valid Syslog severity textual code
081     */
082    public static Severity fromLabel(String label) throws IllegalArgumentException {
083        if (label == null || label.isEmpty())
084            return null;
085
086        Severity severity = severityFromLabel.get(label);
087        if (severity == null) {
088            throw new IllegalArgumentException("Invalid severity '" + label + "'");
089        }
090        return severity;
091    }
092
093    /**
094     * Syslog severity numerical code
095     */
096    public int numericalCode() {
097        return numericalCode;
098    }
099
100    /**
101     * Syslog severity textual code. Not {@code null}.
102     */
103    public String label() {
104        return label;
105    }
106
107    /**
108     * Compare on {@link Severity#numericalCode()}
109     */
110    public static Comparator<Severity> comparator() {
111        return new Comparator<Severity>() {
112            @Override
113            public int compare(Severity s1, Severity s2) {
114                return Integer.compare(s1.numericalCode, s2.numericalCode);
115            }
116        };
117    }
118}
