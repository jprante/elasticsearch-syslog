001package org.elasticsearch.common.syslog;
002
003import org.elasticsearch.ElasticsearchIllegalArgumentException;
004import org.elasticsearch.common.cache.Cache;
005import org.elasticsearch.common.cache.CacheBuilder;
006import org.elasticsearch.common.cache.CacheLoader;
007import org.elasticsearch.common.joda.Joda;
008import org.elasticsearch.common.joda.time.DateTime;
009import org.elasticsearch.common.joda.time.format.DateTimeFormat;
010import org.elasticsearch.common.joda.time.format.DateTimeFormatter;
011import org.elasticsearch.common.xcontent.XContentBuilder;
012
013import java.io.IOException;
014import java.util.Locale;
015import java.util.regex.Pattern;
016
017/**
018 * Parses a syslog message with RFC 3164 or RFC 5424 date format
019 */
020public class MessageParser {
021
022    private final static Pattern TWO_SPACES = Pattern.compile("  ");
023
024    private final static DateTimeFormatter formatter =
025            Joda.forPattern("dateOptionalTime", Locale.ROOT).printer();
026
027    private final static DateTimeFormatter rfc3164Format =
028            DateTimeFormat.forPattern("MMM d HH:mm:ss").withZoneUTC();
029
030    private final static int RFC3164_LEN = 15;
031
032    private final static int RFC5424_PREFIX_LEN = 19;
033
034    private final DateTimeFormatter timeParser;
035
036    private Cache<String, Long> timestampCache;
037
038    public MessageParser() {
039        timeParser = DateTimeFormat.forPattern("yyyy-MM-dd'T'HH:mm:ss").withZoneUTC();
040        timestampCache = CacheBuilder.newBuilder().maximumSize(1000).build(
041                new CacheLoader<String, Long>() {
042
043                    @Override
044                    public Long load(String key) throws Exception {
045                        return timeParser.parseMillis(key);
046                    }
047                });
048    }
049
050    public void parseMessage(String msg, XContentBuilder builder) throws IOException {
051        int msgLen = msg.length();
052        int pos = 0;
053        if (msg.charAt(pos) != '<') {
054            throw new ElasticsearchIllegalArgumentException("bad format: invalid priority: cannot find open bracket '<' " + msg);
055        }
056        int end = msg.indexOf('>');
057        if (end < 0 || end > 6) {
058            throw new ElasticsearchIllegalArgumentException("bad format: invalid priority: cannot find end bracket '>' " + msg);
059        }
060        int pri = Integer.parseInt(msg.substring(1, end));
061        Facility facility = Facility.fromNumericalCode(pri / 8);
062        Severity severity = Severity.fromNumericalCode(pri % 8);
063        builder.field("facility", facility.label()).field("severity", severity.label());
064        if (msgLen <= end + 1) {
065            throw new ElasticsearchIllegalArgumentException("bad format: no data except priority " + msg);
066        }
067        pos = end + 1;
068        if (msgLen > pos + 2 && "1 ".equals(msg.substring(pos, pos + 2))) {
069            pos += 2;
070        }
071        long timestamp;
072        char ch = msg.charAt(pos);
073        if (ch == '-') {
074            timestamp = System.currentTimeMillis();
075            if (msgLen <= pos + 2) {
076                throw new ElasticsearchIllegalArgumentException("bad syslog format (missing hostname)");
077            }
078            pos += 2;
079        } else if (ch >= 'A' && ch <= 'Z') {
080            if (msgLen <= pos + RFC3164_LEN) {
081                throw new ElasticsearchIllegalArgumentException("bad timestamp format");
082            }
083            timestamp = parseRFC3164Time(msg.substring(pos, pos + RFC3164_LEN));
084            pos += RFC3164_LEN + 1;
085        } else {
086            int sp = msg.indexOf(' ', pos);
087            if (sp == -1) {
088                throw new ElasticsearchIllegalArgumentException("bad timestamp format");
089            }
090            timestamp = parseRFC5424Date(msg.substring(pos, sp));
091            pos = sp + 1;
092        }
093        builder.field("timestamp", formatter.print(timestamp));
094        int ns = msg.indexOf(' ', pos);
095        if (ns == -1) {
096            throw new ElasticsearchIllegalArgumentException("bad syslog format (missing hostname)");
097        }
098        String hostname = msg.substring(pos, ns);
099        builder.field("host", hostname);
100        String data;
101        if (msgLen > ns + 1) {
102            pos = ns + 1;
103            data = msg.substring(pos);
104        } else {
105            data = msg;
106        }
107        builder.field("message", data);
108    }
109
110    private Long parseRFC5424Date(String msg) {
111        int len = msg.length();
112        if (len <= RFC5424_PREFIX_LEN) {
113            throw new ElasticsearchIllegalArgumentException("bad format: not a valid RFC5424 timestamp: " + msg);
114        }
115        String timestampPrefix = msg.substring(0, RFC5424_PREFIX_LEN);
116        Long timestamp = timestampCache.getIfPresent(timestampPrefix);
117        int pos = RFC5424_PREFIX_LEN;
118        if (timestamp == null) {
119            throw new ElasticsearchIllegalArgumentException("parse error: timestamp is null");
120        }
121        if (msg.charAt(pos) == '.') {
122            boolean found = false;
123            int end = pos + 1;
124            if (len <= end) {
125                throw new ElasticsearchIllegalArgumentException("bad timestamp format (no TZ)");
126            }
127            while (!found) {
128                char ch = msg.charAt(end);
129                if (ch >= '0' && ch <= '9') {
130                    end++;
131                } else {
132                    found = true;
133                }
134            }
135            if (end - (pos + 1) > 0) {
136                long milliseconds = (long) (Double.parseDouble(msg.substring(pos, end)) * 1000.0);
137                timestamp += milliseconds;
138            } else {
139                throw new ElasticsearchIllegalArgumentException("bad format: invalid timestamp (fractional portion): " + msg);
140            }
141            pos = end;
142        }
143        char ch = msg.charAt(pos);
144        if (ch != 'Z') {
145            if (ch == '+' || ch == '-') {
146                if (len <= pos + 5) {
147                    throw new ElasticsearchIllegalArgumentException("bad format: invalid timezone: " + msg);
148                }
149                int sign = ch == '+' ? +1 : -1;
150                char[] hourzone = new char[5];
151                for (int i = 0; i < 5; i++) {
152                    hourzone[i] = msg.charAt(pos + 1 + i);
153                }
154                if (hourzone[0] >= '0' && hourzone[0] <= '9'
155                        && hourzone[1] >= '0' && hourzone[1] <= '9'
156                        && hourzone[2] == ':'
157                        && hourzone[3] >= '0' && hourzone[3] <= '9'
158                        && hourzone[4] >= '0' && hourzone[4] <= '9') {
159                    int hourOffset = Integer.parseInt(msg.substring(pos + 1, pos + 3));
160                    int minOffset = Integer.parseInt(msg.substring(pos + 4, pos + 6));
161                    timestamp -= sign * ((hourOffset * 60) + minOffset) * 60000;
162                } else {
163                    throw new ElasticsearchIllegalArgumentException("bad format: invalid timezone: " + msg);
164                }
165            }
166        }
167        return timestamp;
168    }
169
170    private Long parseRFC3164Time(String timestamp) {
171        DateTime now = DateTime.now();
172        int year = now.getYear();
173        timestamp = TWO_SPACES.matcher(timestamp).replaceFirst(" ");
174        DateTime date;
175        try {
176            date = rfc3164Format.parseDateTime(timestamp);
177        } catch (Exception e) {
178            return 0L;
179        }
180        if (date != null) {
181            DateTime fixed = date.withYear(year);
182            if (fixed.isAfter(now) && fixed.minusMonths(1).isAfter(now)) {
183                fixed = date.withYear(year - 1);
184            } else if (fixed.isBefore(now) && fixed.plusMonths(1).isBefore(now)) {
185                fixed = date.withYear(year + 1);
186            }
187            date = fixed;
188        }
189        return date == null ? 0L : date.getMillis();
190    }
